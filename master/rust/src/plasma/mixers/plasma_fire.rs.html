use crate::{color::*, mixer::*, phase_amp::*, simd_polyfill::*};
use derive_more::*;
use std::{marker::PhantomData, f32::consts::PI, ops::*};

pub type PlasmaICPFire<'a> = PlasmaInterCalcProducerFire<'a, [PhaseAmp]>;
pub type PlasmaICPFireExtPa<'a> = PlasmaInterCalcProducerFire<'a, [f32]>;

/// A default implementation of a [Mixer] is provided for this struct.
#[derive(Debug, Copy, Clone, PartialEq, Constructor)]
pub struct PlasmaMixerFire;

/// Provides a default implementation of a [IntermediateCalculator].
pub struct PlasmaLineCalcFireH {
    amplitude: Flt,
    phase:     Flt,
    factor:    Flt,
}

/// Provides a default implementation of a [IntermediateCalculator].
pub struct PlasmaLineCalcFireV {
    amplitude: Flt,
    phase:     Flt,
    factor:    Flt,
}

/// Provides a default implementation of an iterator of [PlasmaLineCalcFireH].
pub struct PlasmaMixIterFireH<'a, P: PhaseAmpsSelect<'a> + ?Sized> {
    pa_iter: std::iter::Enumerate<<P as PhaseAmpsSelect<'a>>::IterOne>,
}

/// Provides a default implementation of an iterator of [PlasmaLineCalcFireH].
pub struct PlasmaMixIterFireV<'a, P: PhaseAmpsSelect<'a> + ?Sized> {
    pa_iter: std::iter::Enumerate<<P as PhaseAmpsSelect<'a>>::IterOne>,
}

/// Provides a default implementation of a [IntermediateCalculatorProducer].
pub struct PlasmaInterCalcProducerFire<'a, P: 'a + PhaseAmpsSelect<'a> + ?Sized>(PhantomData<&'a P>);

impl<'a, P> ExactSizeIterator for PlasmaMixIterFireH<'a, P> where P: PhaseAmpsSelect<'a> + ?Sized
{
    #[inline]
    fn len(&self) -> usize { self.pa_iter.len() }
}

impl<'a, P> Iterator for PlasmaMixIterFireH<'a, P> where P: PhaseAmpsSelect<'a> + ?Sized
{
    type Item = PlasmaLineCalcFireH;

    fn next(&mut self) -> Option<Self::Item> {
        match self.pa_iter.next() {
            Some((i, pa)) => {
                let factor = splat(((i % 2) + 1) as f32);
                Some(PlasmaLineCalcFireH { amplitude: splat(pa.amplitude()),
                                         phase:     splat(pa.phase()),
                                         factor:    factor, })
            },
            None => None,
        }
    }

    #[inline]
    fn size_hint(&self) -> (usize, Option<usize>) { self.pa_iter.size_hint() }
}

impl<'a, P> ExactSizeIterator for PlasmaMixIterFireV<'a, P> where P: PhaseAmpsSelect<'a> + ?Sized
{
    #[inline]
    fn len(&self) -> usize { self.pa_iter.len() }
}

impl<'a, P> Iterator for PlasmaMixIterFireV<'a, P> where P: PhaseAmpsSelect<'a> + ?Sized
{
    type Item = PlasmaLineCalcFireV;

    fn next(&mut self) -> Option<Self::Item> {
        match self.pa_iter.next() {
            Some((i, pa)) => {
                let factor = splat(((i % 4) + 1) as f32);
                Some(PlasmaLineCalcFireV { amplitude: splat(pa.amplitude()),
                                         phase:     splat(pa.phase()),
                                         factor:    factor, })
            },
            None => None,
        }
    }

    #[inline]
    fn size_hint(&self) -> (usize, Option<usize>) { self.pa_iter.size_hint() }
}

impl<'a, P> IntermediateCalculatorProducer<'a, P, Flt> for PlasmaInterCalcProducerFire<'a, P>
    where P: PhaseAmpsSelect<'a> + ?Sized
{
    type CalcIterH = PlasmaMixIterFireH<'a, P>;
    type CalcIterV = PlasmaMixIterFireV<'a, P>;
    type LineCalcH = PlasmaLineCalcFireH;
    type LineCalcV = PlasmaLineCalcFireV;

    fn compose_h_iter(pa: &'a P) -> Self::CalcIterH {
        PlasmaMixIterFireH { pa_iter: pa.select(0..12).iter_phase_amps().enumerate() }
    }

    fn compose_v_iter(pa: &'a P) -> Self::CalcIterV {
        PlasmaMixIterFireV { pa_iter: pa.select(12..24).iter_phase_amps().enumerate() }
    }
}

        let val1 = compose8(vxp[4], vxp[5], vxp[ 6], vxp[ 7], vyp[4], vyp[5], vyp[ 6], vyp[ 7]);

        let rgb0 = PixelRgb::from_hsv(splat(0.0), splat(1.0), (val1+val2).abs().min(splat(1.0)));

        next_pixel(rgb0+rgb1);

#[inline]
fn compose8(x1: Flt, x2: Flt, x3: Flt, x4: Flt, y1: Flt, y2: Flt, y3: Flt, y4: Flt) -> Flt {
    const NORM: Flt = splat(1.0 / 4.0);
    (x1 + y1 * x2 + y2 * x3 + y3 * x4 + y4) * NORM
}

impl IntermediateCalculator<Flt> for PlasmaLineCalcFireH {
    #[inline]
    fn calculate(&self, v: Flt) -> Flt {
        (v + self.phase).sin() * self.amplitude
    }
}

impl IntermediateCalculator<Flt> for PlasmaLineCalcFireV {
    #[inline]
    fn calculate(&self, v: Flt) -> Flt {
        let vn = v * splat(2.0/PI);
        let v = ((v*self.factor + self.phase) / (vn*splat(2.0))).sin() * self.amplitude * vn;
        v*v
    }
}
